nodemailer.Transporter | null = null;\n\nfunction getTransporter() {\n  if (transporter) return transporter;\n\n  transporter = nodemailer.createTransport({\n    host: process.env.SMTP_HOST || \"localhost\",\n    port: parseInt(process.env.SMTP_PORT || \"587\"),\n    secure: false,\n    auth: process.env.SMTP_USER\n      ? {\n          user: process.env.SMTP_USER,\n          pass: process.env.SMTP_PASSWORD,\n        }\n      : undefined,\n  });\n\n  return transporter;\n}\n\nexport async function sendEmail(\n  to: string,\n  subject: string,\n  html: string\n): Promise<boolean> {\n  try {\n    await getTransporter().sendMail({ to, subject, html });\n    return true;\n  } catch (error) {\n    console.error(\"Email error:\", error);\n    return false;\n  }\n}\n```\n\n## Paso
